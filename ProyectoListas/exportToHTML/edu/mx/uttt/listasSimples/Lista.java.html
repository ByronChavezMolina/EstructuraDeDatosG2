package edu.mx.uttt.listasSimples;
import edu.mx.uttt.listasSimples.NodoLista;
public class Lista {
    // Variables de instancia
    private String nombreLista;
    protected NodoLista primerNodo;
    private NodoLista ultimoNodo;

    // Constructor vacío (sin parámetros)
    public Lista() {
        this("Lista Simple");
    }

    // Constructor con parámetros
    public Lista(String nombreLista) {
        this.nombreLista = nombreLista;
        primerNodo = ultimoNodo = null;
    }

    // Método para verificar si la lista está vacía
    public boolean estaVacia() {
        return primerNodo == null;
    }

    // Insertar al frente
    public void insertarAlFrente(int dato) {
        if (estaVacia()) {
            primerNodo = ultimoNodo = new NodoLista(dato);
        } else {
            primerNodo = new NodoLista(dato, primerNodo);
        }
    }

    // Insertar al final
    public void insertarAlFinal(int dato) {
        if (estaVacia()) {
            primerNodo = ultimoNodo = new NodoLista(dato);
        } else {
            ultimoNodo = ultimoNodo.sigNodo = new NodoLista(dato);
        }
    }

    // Imprimir la lista
    public void imprimir() {
        if (estaVacia()) {
            System.out.println("La lista está vacía");
        } else {
            NodoLista aux = primerNodo;
            while (aux != null) {
                System.out.print(aux.dato + "->");
                aux = aux.sigNodo;
            }
            System.out.println();
        }
    }


    // Insertar en una posición específica
    public void insertarEnPosicion(int dato, int posicion) {
        if (posicion <= 0 || estaVacia()) {
            insertarAlFrente(dato);
        } else {
            NodoLista actual = primerNodo;
            int contador = 0;

            while (actual != null && contador < posicion - 1) {
                actual = actual.sigNodo;
                contador++;
            }

            if (actual == null || actual.sigNodo == null) {
                insertarAlFinal(dato);
            } else {
                actual.sigNodo = new NodoLista(dato, actual.sigNodo);
            }
        }
    }

    // Eliminar del frente
    public int eliminarDelFrente() {
        int datoAEliminar = -1;
        if (!estaVacia()) {
            datoAEliminar = primerNodo.dato;
            if (primerNodo == ultimoNodo) {
                primerNodo = ultimoNodo = null;
            } else {
                primerNodo = primerNodo.sigNodo;
            }
        }
        return datoAEliminar;
    }

    // Eliminar al final
    public int eliminarAlFinal() {
        int datoAEliminar = -1;
        if (!estaVacia()) {
            if (primerNodo == ultimoNodo) {
                datoAEliminar = primerNodo.dato;
                primerNodo = ultimoNodo = null;
            } else {
                NodoLista actual = primerNodo;
                while (actual.sigNodo != ultimoNodo) {
                    actual = actual.sigNodo;
                }
                datoAEliminar = ultimoNodo.dato;
                ultimoNodo = actual;
                ultimoNodo.sigNodo = null;
            }
        }
        return datoAEliminar;
    }

    // Eliminar en una posición específica
    public int eliminarEnPosicion(int posicion) {
        if (posicion <= 0 || estaVacia()) {
            return eliminarDelFrente();
        }

        NodoLista actual = primerNodo;
        NodoLista previo = null;
        int contador = 0;

        while (actual != null && contador < posicion) {
            previo = actual;
            actual = actual.sigNodo;
            contador++;
        }

        if (actual == null) {
            return -1; // Posición no válida
        } else if (actual == ultimoNodo) {
            return eliminarAlFinal();
        } else {
            previo.sigNodo = actual.sigNodo;
            return actual.dato;
        }
    }

    // Buscar un dato
    public boolean buscar(int dato) {
        NodoLista actual = primerNodo;
        while (actual != null) {
            if (actual.dato == dato) {
                return true;
            }
            actual = actual.sigNodo;
        }
        return false;
    }

    // Ordenamiento usando el método burbuja
    public void ordenarBurbuja() {
        if (!estaVacia()) {
            boolean intercambio;
            do {
                intercambio = false;
                NodoLista actual = primerNodo;

                while (actual != null && actual.sigNodo != null) {
                    if (actual.dato > actual.sigNodo.dato) {
                        // Intercambiar datos
                        int temp = actual.dato;
                        actual.dato = actual.sigNodo.dato;
                        actual.sigNodo.dato = temp;

                        intercambio = true;
                    }
                    actual = actual.sigNodo;
                }
            } while (intercambio);
        }
    }
}
